113 void cppp_accelerationNmiCallback(void)
114 {
115  static int16_t temp;
116 
117  switch(cppp_operationMode){
118  case NORMAL_OPERATION_MODE:
119  cppp_operationMode = CALIBRATION_READ_POSITIVE_X_MODE;
120  break;
121  case CALIBRATION_READ_POSITIVE_X_MODE:
122  cppp_operationMode = CALIBRATION_READ_NEGATIVE_X_MODE;
123  temp = cppp_accelerationData[ACCELERATION_AXIS_X];
124  break;
125  case CALIBRATION_READ_NEGATIVE_X_MODE:
126  cppp_operationMode = CALIBRATION_READ_POSITIVE_Y_MODE;
127  cppp_sensitivity[ACCELERATION_AXIS_X] = (temp - cppp_accelerationData[ACCELERATION_AXIS_X])>>1;
128  break;
129  case CALIBRATION_READ_POSITIVE_Y_MODE:
130  cppp_operationMode = CALIBRATION_READ_NEGATIVE_Y_MODE;
131  temp = cppp_accelerationData[ACCELERATION_AXIS_Y];
132  break;
133  case CALIBRATION_READ_NEGATIVE_Y_MODE:
134  cppp_operationMode = CALIBRATION_READ_POSITIVE_Z_MODE;
135  cppp_sensitivity[ACCELERATION_AXIS_Y] = (temp - cppp_accelerationData[ACCELERATION_AXIS_Y])>>1;
136  break;
137  case CALIBRATION_READ_POSITIVE_Z_MODE:
138  cppp_operationMode = CALIBRATION_READ_NEGATIVE_Z_MODE;
139  temp = cppp_accelerationData[ACCELERATION_AXIS_Z];
140  break;
141  case CALIBRATION_READ_NEGATIVE_Z_MODE:
142  cppp_operationMode = NORMAL_OPERATION_MODE;
143  cppp_sensitivity[ACCELERATION_AXIS_Z] = (temp - cppp_accelerationData[ACCELERATION_AXIS_Z])>>1;
144  break;
